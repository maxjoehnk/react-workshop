=== Mobx

[,typescript]
----
class TodoState {
	todos: Todo[] = []

	constructor() {
		makeAutoObservable(this)
	}

  async fetchTodos() {
		const result = await loadTodos();
		runInAction(() => {
			this.todos = result;
		});
	}

	addTodo(todo: Todo) {
		this.todos.push(todo)
	}
}
----

=== Mobx

[,typescript]
----
import { observer } from "mobx-react-lite"

const App = observer(({ todosState }) => {
	useEffect(() => {
		todosState.fetchTodos();
	}, []);

	return <div>
		{todosState.todos.map(todo => <TodoItem todo={todo} />)}
	</div>
});
