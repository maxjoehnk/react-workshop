=== Testing

[,typescript]
----
import { screen, render } from '@testing-library/react';
import { IssueType, IssueTypeName } from './issue-type-name.tsx';

describe('IssueTypeName', () => {
	test.each([
		[IssueType.Story, 'Story'],
		[IssueType.Bug, 'Fehler'],
	])('should translate issue type', (issueType: IssueType, expected: string) => {
		render(<IssueTypeName issueType={issueType} />);

		expect(screen.getByText(expected)).toBeInTheDocument();
	});
});
----

=== Testing

[,typescript]
----
import UserActions from './user-actions.tsx';
import { render, screen } from '@testing-library/react';
import { userEvent } from '@testing-library/user-event';

describe('UserActions', () => {
	test('clicking Delete should call onDelete', async () => {
		const callback = vi.fn();
		render(<UserActions onDelete={callback} />);

		await userEvent.click(screen.getByText('Löschen'));

		expect(callback).toHaveBeenCalledOnce();
	});
});
